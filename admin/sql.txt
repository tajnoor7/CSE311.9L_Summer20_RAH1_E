CREATE TABLE admin(
    name varchar(50),
    email varchar(50),
    password varchar(50)
)


CREATE TABLE booking (
    passenger_fname varchar(50),
    passenger_lname varchar(50),
    passenger_email varchar(50),
    travel_from varchar(50),
    travel_to varchar(50),
    date date,
    seat varchar(50),
    class varchar(50)
)


CREATE TABLE route(
    train_id int(50),
    train_name varchar(50),
    train_route varchar(50)
)


CREATE TABLE ticket(
    train_name varchar(50),
    economy_class_price int(50),
    business_class_price int(50)
)


CREATE TABLE train(
    train_id int(50),
    train_name varchar(50),
    departure_from varchar(50),
    arrival_to varchar(50),
    departure_time Varchar(50)
)


CREATE TABLE user(
    name varchar(50),
    email varchar(50),
    password varchar(50)
)
